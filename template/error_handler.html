Error Handler

1. Create a file named applicationError.js in the middleware folder.

2. Copy the following code in the file. 


        (i) applicationError.js

        
        export class ApplicationError extends Error{
            constructor(message, code){
                super(message);
                this.code = code;
            }
        }
         
    
        
        (ii) import { ApplicationError } from './src/error-handler/applicationError.js';

        (iii) // Error handler middleware
            server.use((err, req, res, next)=>{
            console.log(err);
            if (err instanceof ApplicationError){
                res.status(err.code).send(err.message);
            }

        (iv) // server errors.
        res.status(500).send('Something went wrong, please try later' );
        });
    
            

        (v) // Middleware to handle 404 requests.
        server.use((req, res)=>{
        res.status(404).send("API not found. Please check our documentation for more information at localhost:3200/api-docs")
        });
